\section*{2. Economic Model}

\textbf{Annual Material Savings}

\[
M = 12 \times T \times C \times \Delta S
\]

Where:

\begin{itemize}
\item $T$ = tons per month
\item $C$ = steel cost per ton
\item $\Delta S$ = scrap improvement delta
\end{itemize}

\textbf{Optional Labor Efficiency}

\[
L = 12 \times T \times \Delta H \times W
\]

\[
V = M + L
\]

\section*{3. Example — Commercial Fabrication Yard}

Assumptions:

\begin{itemize}
\item 60 tons/month
\item \$900/ton steel
\item 1.5 pt scrap delta
\item 0.10 hr/ton labor reduction
\item \$85/hr labor rate
\end{itemize}

Material savings:

\[
M = 12 \times 60 \times 900 \times 0.015 = \$9{,}720
\]

Labor savings:

\[
L = 12 \times 60 \times 0.10 \times 85 = \$6{,}120
\]

Total modeled value:

\[
V = \$15{,}840
\]

\section*{4. ROI Comparison}

If subscription capture = 15\%:

\[
P = 0.15 \times 15{,}840 = \$2{,}376
\]

Net gain:

\[
15{,}840 - 2{,}376 = 13{,}464
\]

Annual ROI:

\[
\frac{13{,}464}{2{,}376} = 5.66 = 566\%
\]
